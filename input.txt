# Suma
suma = 2 + 3
print(suma)

# Resta
resta = 5 - 2
print(resta)

# Multiplicación
multiplicacion = 3 * 4
print(multiplicacion)

# División
division = 10 / 2
print(division)

# Condicionales
numero = 10

if numero > 0:
    print("El número es positivo")
else:
    print("El número es negativo")

# Bucles
for i in range(5):
    print(i)

# Funciones
def suma(a, b):
    return a + b

resultado = suma(2, 3)
print(resultado)
